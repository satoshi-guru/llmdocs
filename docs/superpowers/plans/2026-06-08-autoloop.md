# autoloop Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build `autoloop` — a Python binary that takes one goal and self-drives plan→fan-out→verify→re-plan on the Claude Agent SDK until the goal is met or a budget is hit, with no re-invocation.

**Architecture:** A thin evaluator-optimizer shell in an isolated `autoloop/` package. `ClaudeSDKClient` runs a continuous session; each turn Claude plans + fans out to subagents and emits a structured `Verdict`; the driver evaluates the stop condition, accounts cost, enforces a `PreToolUse` guardrail, and persists `session_id` for resume. Reuses the existing fan-out-code / agent-patterns orchestration vocabulary; does NOT import the doc-pipeline code.

**Tech Stack:** Python 3.10+, `claude-agent-sdk`, `pydantic` (both an optional `[autoloop]` extra), `pytest`. Spec: `docs/superpowers/specs/2026-06-08-autoloop-design.md`.

**Branch:** work on `feat/autoloop` (already exists, holds the spec). Commit + push each task; never on `main`.

```python
# autoloop/schemas.py
"""Pydantic models used as Agent SDK output_format schemas and typed returns."""
from __future__ import annotations

from typing import Literal
from pydantic import BaseModel, Field


class Task(BaseModel):
    id: str
    title: str
    prompt: str
    agent: str
    status: Literal["pending", "done", "failed"] = "pending"


class Plan(BaseModel):
    rationale: str
    tasks: list[Task] = Field(default_factory=list)


class Verdict(BaseModel):
    """The per-turn structured output the driver reads to decide stop/continue."""
    done: bool
    reason: str
    remaining: list[str] = Field(default_factory=list)


class RunResult(BaseModel):
    status: Literal["done", "budget", "maxturns", "error"]
    turns: int
    cost_usd: float
    session_id: str | None
    reason: str
    log_path: str
```

```python
# autoloop/guardrail.py
"""PreToolUse hook enforcing the unattended deny set.

Return shape per claude-agent-sdk/en/agent-sdk/hooks.md: deny with a
hookSpecificOutput.permissionDecision == "deny"; allow with {}.
"""
from __future__ import annotations

import re
from pathlib import Path
from typing import Callable

# Bash command substrings/patterns that are never allowed unattended.
_BASH_DENY = [
    r"\brm\s+-rf\b",
    r"git\s+push\s+.*(--force|-f)\b",
    r"git\s+reset\s+--hard\b",
    r"\bssh\s+\S+",
    r"\bscp\s+\S+",
    r"\bdeploy\b",
    r"\bsystemctl\b",
    r"\bsudo\b",
    r"(curl|wget)\s+.*\|\s*(sh|bash)\b",
]

# Path fragments that may never be written/edited.
_PATH_DENY = [".env", ".pem", "id_rsa", "id_ed25519", "/.ssh/"]


def _deny(event_name: str, reason: str) -> dict:
    return {"hookSpecificOutput": {
        "hookEventName": event_name,
        "permissionDecision": "deny",
        "permissionDecisionReason": reason,
    }}


def make_guardrail(extra_deny: list[str] | None, cwd: Path) -> Callable:
    cwd = cwd.resolve()
    bash_pats = [re.compile(p) for p in _BASH_DENY]
    extra = [re.compile(re.escape(x)) for x in (extra_deny or [])]

    async def guardrail(input_data, tool_use_id, context):
        event = input_data.get("hook_event_name", "PreToolUse")
        tool = input_data.get("tool_name", "")
        args = input_data.get("tool_input", {}) or {}

        if tool == "Bash":
            cmd = args.get("command", "") or ""
            for pat in bash_pats + extra:
                if pat.search(cmd):
                    return _deny(event, f"blocked by autoloop guardrail: {pat.pattern!r}")
            return {}

        if tool in ("Write", "Edit", "NotebookEdit"):
            path = args.get("file_path", "") or ""
            low = path.lower()
            if any(frag in low for frag in _PATH_DENY):
                return _deny(event, f"refuse to write protected path: {path}")
            # Refuse writes outside the working tree.
            try:
                Path(path).resolve().relative_to(cwd)
            except ValueError:
                return _deny(event, f"refuse to write outside cwd: {path}")
            return {}

        return {}

    return guardrail
```

```python
"""Fan-out worker registry for the autoloop driver."""
from __future__ import annotations


def build_registry(overrides: dict | None = None) -> dict:
    """Return {name: AgentDefinition}. `overrides` replace/add by name."""
    from claude_agent_sdk import AgentDefinition

    registry = {
        "fixer": AgentDefinition(
            description="Implements one assigned task and runs its test.",
            prompt=(
                "You implement exactly the task you are given. Make the change, run the "
                "most relevant test, and report what you changed (files + a one-line "
                "summary) and the test result. Do not touch unrelated code."
            ),
            tools=["Read", "Edit", "Bash", "Grep", "Glob"],
        ),
        "verifier": AgentDefinition(
            description="Independently verifies a claimed fix. Read-only.",
            prompt=(
                "You verify a claim independently. Re-run the relevant test or inspect the "
                "code yourself. Report PASS or FAIL with the exact evidence (command + "
                "output excerpt). Do not edit files."
            ),
            tools=["Read", "Bash", "Grep", "Glob"],
        ),
        "researcher": AgentDefinition(
            description="Answers a focused question from the repo + llmdocs store.",
            prompt=(
                "Answer the focused question using the repository and the llmdocs store "
                "(grep ~/.llmdocs/docs/<lib>/LOOKUP.md before reading pages). Cite "
                "file:line. Be terse; return only the answer."
            ),
            tools=["Read", "Grep", "Glob", "Bash"],
        ),
    }
    if overrides:
        registry.update(overrides)
    return registry
```

```python
import asyncio
from pathlib import Path
from autoloop.driver import run


class _Result:
    def __init__(self, structured, cost=0.1, sid="sess-1"):
        self.structured_output = structured
        self.total_cost_usd = cost
        self.session_id = sid


class _FakeClient:
    """Scripts a sequence of per-turn Verdict dicts."""
    def __init__(self, scripted, prompts):
        self._scripted = list(scripted)
        self._prompts = prompts  # shared list to record prompts
        self.options = None

    async def __aenter__(self): return self
    async def __aexit__(self, *a): return False

    async def query(self, prompt, session_id="default"):
        self._prompts.append(prompt)

    async def receive_response(self):
        verdict, cost = self._scripted.pop(0)
        yield _Result(verdict, cost=cost)


def _factory(scripted, prompts):
    def make(options=None):
        c = _FakeClient(scripted, prompts)
        c.options = options
        return c
    return make
```

```python
# autoloop/driver.py
"""The self-driving loop: plan→fan-out→verify→re-plan until done or budget."""
from __future__ import annotations

import json
import os
import time
from pathlib import Path
from typing import Callable

from autoloop.agents import build_registry
from autoloop.guardrail import make_guardrail
from autoloop.schemas import RunResult, Verdict

OPERATING_RULES = (
    "You are an autonomous engineer driving ONE goal to completion across multiple turns.\n"
    "Each turn: break the outstanding work into independent units and delegate them to the "
    "named subagents (fixer/verifier/researcher) via the Agent tool, in parallel where "
    "independent. Always have a fix verified by the verifier before counting it done.\n"
    "Never run deploy, force-push, ssh, or destructive commands.\n"
    "End every turn by returning the Verdict: set done=true ONLY when the whole goal is met "
    "AND verified; otherwise list what is still remaining."
)


def _log_path(goal: str) -> Path:
    home = Path(os.environ.get("LLMDOCS_HOME", Path.home() / ".llmdocs"))
    d = home / "autoloop"
    d.mkdir(parents=True, exist_ok=True)
    slug = "".join(c if c.isalnum() else "-" for c in goal.lower())[:40].strip("-")
    return d / f"{int(time.time())}-{slug}.jsonl"


def _turn_prompt(goal: str, last: Verdict | None) -> str:
    if last is None:
        return f"GOAL: {goal}\n\nBegin. Plan and start the work, then return the Verdict."
    return ("Continue toward the goal. Outstanding items:\n- "
            + "\n- ".join(last.remaining or ["(none listed — re-assess)"])
            + "\n\nDo the next round of work, then return the Verdict.")


def _build_options(goal, budget_usd, max_turns, model, resume, dry_run, cwd, registry):
    from claude_agent_sdk import ClaudeAgentOptions, HookMatcher
    guardrail = make_guardrail(extra_deny=None, cwd=cwd)
    return ClaudeAgentOptions(
        system_prompt={"type": "preset", "preset": "claude_code", "append": OPERATING_RULES},
        permission_mode="plan" if dry_run else "acceptEdits",
        agents=registry or build_registry(None),
        hooks={"PreToolUse": [HookMatcher(matcher="Bash|Write|Edit", hooks=[guardrail])]},
        max_turns=max_turns,
        max_budget_usd=budget_usd,
        output_format={"type": "json_schema", "schema": Verdict.model_json_schema()},
        setting_sources=["project"],
        allowed_tools=["Agent", "Read", "Edit", "Bash", "Grep", "Glob"],
        model=model,
        resume=resume,
        cwd=str(cwd),
    )


def _default_factory(options=None):
    from claude_agent_sdk import ClaudeSDKClient
    return ClaudeSDKClient(options=options)


async def run(goal: str, *, budget_usd: float, max_turns: int,
              model: str | None = None, resume: str | None = None,
              dry_run: bool = False, cwd: Path = None,
              registry: dict | None = None,
              client_factory: Callable | None = None,
              log: Callable[[str], None] = lambda m: None) -> RunResult:
    cwd = (cwd or Path.cwd()).resolve()
    factory = client_factory or _default_factory
    # Real SDK needs options; the fake factory ignores them, so building is safe either way.
    try:
        options = _build_options(goal, budget_usd, max_turns, model, resume, dry_run, cwd, registry)
    except Exception:
        options = None  # fake factory path (no SDK installed in unit tests)

    log_path = _log_path(goal)
    cost = 0.0
    turns = 0
    session_id = resume
    last: Verdict | None = None

    try:
        async with factory(options=options) as client:
            while True:
                await client.query(_turn_prompt(goal, last))
                result = None
                async for message in client.receive_response():
                    if getattr(message, "structured_output", None) is not None:
                        result = message
                turns += 1
                cost += float(getattr(result, "total_cost_usd", 0.0) or 0.0)
                session_id = getattr(result, "session_id", None) or session_id
                verdict = Verdict.model_validate(getattr(result, "structured_output", {}) or {})
                _append_log(log_path, turns, verdict, cost)
                log(f"turn {turns}: done={verdict.done} cost=${cost:.2f} — {verdict.reason}")

                if verdict.done:
                    return RunResult(status="done", turns=turns, cost_usd=cost,
                                     session_id=session_id, reason=verdict.reason,
                                     log_path=str(log_path))
                if cost >= budget_usd:
                    return RunResult(status="budget", turns=turns, cost_usd=cost,
                                     session_id=session_id, reason="budget exhausted",
                                     log_path=str(log_path))
                if turns >= max_turns:
                    return RunResult(status="maxturns", turns=turns, cost_usd=cost,
                                     session_id=session_id, reason="turn cap reached",
                                     log_path=str(log_path))
                last = verdict
    except Exception as exc:  # terminal API error etc. — never crash mid-loop
        return RunResult(status="error", turns=turns, cost_usd=cost,
                         session_id=session_id, reason=str(exc), log_path=str(log_path))


def _append_log(path: Path, turn: int, verdict: Verdict, cost: float) -> None:
    rec = {"turn": turn, "done": verdict.done, "reason": verdict.reason,
           "remaining": verdict.remaining, "cost_usd": round(cost, 4)}
    with path.open("a", encoding="utf-8") as f:
        f.write(json.dumps(rec) + "\n")
```

```python
# autoloop/cli.py
"""Command-line entry point for autoloop."""
from __future__ import annotations

import argparse
import asyncio
import os
import sys
from pathlib import Path


def build_parser() -> argparse.ArgumentParser:
    p = argparse.ArgumentParser(
        prog="autoloop",
        description="Self-drive ONE goal to completion on the Claude Agent SDK.")
    p.add_argument("goal", help="the goal to drive to completion")
    p.add_argument("--budget", type=float, default=5.0,
                   help="soft USD ceiling (estimate, not a bill); default 5.00")
    p.add_argument("--max-turns", type=int, default=20, help="hard turn cap; default 20")
    p.add_argument("--model", default=None, help="model id override")
    p.add_argument("--resume", default=None, help="resume a prior run's session_id")
    p.add_argument("--dry-run", action="store_true",
                   help="plan only, zero writes (permission_mode=plan)")
    p.add_argument("--cwd", default=None, help="working directory (default: current)")
    return p


def main(argv: list[str] | None = None) -> int:
    args = build_parser().parse_args(argv)
    if not os.environ.get("ANTHROPIC_API_KEY"):
        print("error: ANTHROPIC_API_KEY is not set — autoloop needs API access.",
              file=sys.stderr)
        return 2

    from autoloop.driver import run
    print("note: --budget is a client-side cost estimate, not authoritative billing.",
          file=sys.stderr)
    result = asyncio.run(run(
        args.goal, budget_usd=args.budget, max_turns=args.max_turns,
        model=args.model, resume=args.resume, dry_run=args.dry_run,
        cwd=Path(args.cwd) if args.cwd else None,
        log=lambda m: print(m, file=sys.stderr),
    ))
    print(f"\n{result.status.upper()} after {result.turns} turn(s), "
          f"~${result.cost_usd:.2f}. session={result.session_id}\n{result.reason}\n"
          f"log: {result.log_path}")
    return 0 if result.status in ("done", "budget", "maxturns") else 1


if __name__ == "__main__":
    raise SystemExit(main())
```
